document.addEventListener("DOMContentLoaded", function ()
 {

    const completed = {{ completed | default(0) | int }};
    const pending = {{ pending | default(0) | int }};
    const accepted = {{ accepted | default(0) | int }};
    const rejected = {{ rejected | default(0) | int }};
    const in_transit = {{ in_transit | default(0) | int }};

    const categories = {{ categories | default([]) | tojson }};
    const categoryValues = {{ category_values | default([]) | tojson }};

    // PIE CHART
    const statusCtx = document.getElementById('statusChart');
    if (statusCtx) {
        new Chart(statusCtx.getContext('2d'), {
            type: 'pie',
            data: {
                labels: ['Completed', 'Pending', 'Accepted', 'Rejected', 'In Transit'],
                datasets: [{
                    data: [completed, pending, accepted, rejected, in_transit],
                    backgroundColor: [
                        '#4CAF50',
                        '#9E9E9E',
                        '#2196F3',
                        '#F44336',
                        '#FF9800'
                    ]
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false
            }
        });
    }

    // BAR CHART
    const categoryCtx = document.getElementById('categoryChart');
    if (categoryCtx) {
        new Chart(categoryCtx.getContext('2d'), {
            type: 'bar',
            data: {
                labels: categories,
                datasets: [{
                    label: 'Donations',
                    data: categoryValues,
                    backgroundColor: '#6a0dad'
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false
            }
        });
    }


// Notification Dropdown JS
function toggleNotifications() {
    var dropdown = document.getElementById('notificationDropdown');
    dropdown.style.display = dropdown.style.display === 'block' ? 'none' : 'block';
}

// Close dropdown if clicked outside
window.onclick = function(event) {
    if (!event.target.matches('.notification-bell') && !event.target.closest('#notificationDropdown')) {
        document.getElementById('notificationDropdown').style.display = 'none';
    }
}

// Mark notification as read (AJAX)
function markAsRead(notificationId) {
    fetch('/mark_notification_read/' + notificationId, { method: 'POST' })
        .then(response => response.json())
        .then(data => {
            if(data.success) {
                location.reload(); // Reload page to update unread count
            }
        });
}
</script>
